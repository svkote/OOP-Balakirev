Подвиг 2. Имеется следующая программа, в которой объявлены три класса:

class A:
    def __init__(self):
        print("A")
        super().__init__()


class B:
    def __init__(self):
        print("B")
        super().__init__()


class C(A, B):
    def __init__(self):
        print("C")
        super().__init__()

c = C()
